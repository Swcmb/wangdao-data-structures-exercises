1 #include <stdio.h>


  2 #define MaxSize 10  // 定义最大长度


  3 


  4 typedef struct {


  5     int data[MaxSize];  // 用静态的数组存放数据元素


  6     int length;         // 顺序表的当前长度


  7 } SqList;                 // 顺序表的定义


  8 


  9 void InitList(SqList *L) {


 10     for (int i = 0; i < MaxSize; i++) {


 11         L->data[i] = 0;  // 初始化为 0


 12     }


 13     L->length = 0;


 14 }


 15 


 16 int main() {


 17     SqList L;


 18     InitList(&L);


 19     printf("初始化成功！\n");


 20     return 0;


 21 }
